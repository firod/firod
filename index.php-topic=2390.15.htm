  5.          $gray=int((0.299*colorr($c))+(0.587*colorg($c))+(0.114*colorb($c)))
  6.          PXLREPLACE($x,$y,$x,$y,$c,COLORGEN($gray,$gray,$gray))
  7.       END_CYC
  8.    END_CYC
  9. END_SUB

  11.

  15.    $place = $x + $x1*4 - $y1*($xmax+1)*4 //сдвиг в нужное место

  17.    FOR($y=0, $y < ($y2 - $y1 + 1))
  18.

  20.       FOR($x=0, $x < ($x2 - $x1 + 1))

  22.          $gray_canal = INT((0.299*COLORR($c)) + (0.587*COLORG($c)) + (0.114*COLORB($c)))
  23.          $gray = COLORGEN($gray_canal, $gray_canal, $gray_canal)
  24.          WRITEMEM($pid, $adrDECwr + $place, $gray, 4)
