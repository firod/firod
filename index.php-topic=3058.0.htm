твой скрипт выполняется однократно из за HALT.
В кликере можно создавать потоки.

Код: Clickermann
  1. THREAD(t, 1) // код в потоке выполняется параллельно
  2.    LCLICK(RND(1480,1653), RND(877,981))
  3.    WAIT(RND(14,25)
  4. END_THREAD
  5.  
  6. GETSCREEN
  7. IF_PICTURE_IN(0,0,$_xmax,$_ymax,"4.bmp")
  8.    LCLICK(858,741)
  9.    WAITMS(100)
  10. END_IF
  11.  
  12. GETSCREEN
  13. IF_PICTURE_IN(5,3, 1675,909, "3.bmp")
  14.    LCLICK($_return1, $_return2)
  15.    WAIT (5)
  16.    GOTO (aaa)
  17. END_IF
  18.  
  19. GETSCREEN
  20. IF_PICTURE_IN(0,0,$_xmax,$_ymax,"1.bmp")
  21.    LCLICK(1563,929)
  22.    WAIT(77)
  23. END_IF
  24.  
  25. GETSCREEN
  26. IF_PICTURE_IN(0,0,$_xmax,$_ymax,"2.bmp")
  27.    LCLICK(39,418)
  28.    WAITMS(100)
  29.    LCLICK(35,419)
  30.    WAIT(1)
  31.    LCLICK(238,419)
  32.    WAIT(1)
  33.    LCLICK(205, 501)
  34.    WAITMS(100)
  35. END_IF
  36.  
  37. aaa:               //
  38. LCLICK(35,419)
  39. WAIT(1)
  40. LCLICK(238,419)
  41. WAIT(1)
  42. LCLICK(205, 501)
  43. WAIT(1)
  44. LCLICK(522,218)
  45. HALT

THREAD - Начало потока
t - имя потока (можно заменить на любое)
1 - начальное состояние, где 1 - означает рабочее, 0 - приостановленное
END_THREAD - Конец потока. Обязательно ПОСЛЕ действия.
