Если с "Атаковать" неопределенность, то нужно искать сначала "Атаковать" а уж потом если не найдено "Удар"
Цикл FOR не нужен, можно просто продлевать время цикла каждый раз как только картинка найдена.
цикл сам завершится если за 5 секунд ни "Атаковать.bmp" ни "Удар.bmp" не будет найдено.

Код: Clickermann
  1. $time = $_time_t + 5
  2. WHILE($time > $_time_t)
  3.    GETSCREEN(0,0, 350,600)
  4.    IF_PICTURE_IN (0,0, 350,600, "Атаковать.bmp", -1, 100)
  5.       LCLICK($_return1+30, $_return2)
  6.       LOGWRITE ("Атаковать")
  7.       WAITMS(300)
  8.       $time = $_time_t + 5 // Продлеваем время
  9.    ELSE
  10.       IF_PICTURE_IN (0,0, 350,600, "Удар.bmp", -1, 100)
  11.          LCLICK($_return1+20, $_return2)
  12.          LOGWRITE ("Удар")
  13.          WAITMS(300)
  14.          $time = $_time_t + 5 // Продлеваем время
  15.       ELSE
  16.          WAITMS(50)
  17.       END_IF
  18.    END_IF
  19. END_CYC
