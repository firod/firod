Есть необходимость выбора элементов из большого списка.
Написал сценарий. Он выводит большой список элеменов с чекбоксами на экран по 10 штук, до последнего, записывает состоние чекбоксов, создает список выбранных.
1. Может я перемудрил и можно сценарий упростить ?
2. Некорректно работает режим жирного шрифта ($w_flag = 110), где ошибка ?
Код: Clickermann
  1. #include "Window.cms"
  2. //==============================================================================
  3. //LOGCLEAR
  4. $input_file = "Buildings.txt"           // Исходные данные - большой список
  5. $output_file = "Buildings_selected.txt" // Результат - список выбранных
  6. TFREADARR($input_file, $items_all)
  7. TFREADARR("item_flags_all.txt", $itemsflags_all)
  8. TFCLEAR ("item_flags_all.txt", 1)
  9. IF(ARRSIZE($itemsflags_all) < ARRSIZE($items_all))
  10.    FOR($i = ARRSIZE($itemsflags_all) , $i < ARRSIZE($items_all))
  11.       ARRPUSH($itemsflags_all, 0)
  12.    END_CYC
  13. END_IF
  14. //##############################################################################
  15. //Вывод текста на экран по 10 строк с чекбоксами
  16. FOR($a, $a < INT ((ARRSIZE($items_all) / 10)+1))
  17.    $n = $check * 10
  18.    FOR($i = $n, (($i < $n+10) AND ($i < ARRSIZE($items_all)))
  19.       ARRPUSH($items, $items_all[$i])
  20.       ARRPUSH($itemsflags, $itemsflags_all[$i])
  21.    END_CYC
  22.    Window_checkbox_flags(110, "Выберите пункты :", 35)
  23. //   Window_checkbox_flags(110, "Выберите пункты :", 35) //жирным шрифтом текст не помещается ?
  24.    STRSEPARATE ($flags, ",", $item_flags)
  25.    STRSEPARATE($text, "|", $selecteditems)
  26.    $flags = $item_flags[0]
  27.    FOR($i=0, $i < ARRSIZE($item_flags))
  28.       $flags = STRCONCAT($flags, ",", $item_flags[$i])
  29.       IF($item_flags[$i] = 1)                       // Запись выбранных в массив
  30.          ARRPUSH($Selected, $selecteditems[$i]) 
  31.       END_IF
  32.    END_CYC
  33. //==============================================================================
  34. //  Подговка ответа
  35.    IF(ARRSIZE($Selected) > 0)
  36.       $otv = $Selected[0]
  37.       FOR($i=1, $i < ARRSIZE($Selected))
  38.          $otv = STRCONCAT($otv, "\n", $Selected[$i])
  39.       END_CYC
  40.       $text = STRCONCAT("Выбраны пункты:\n\n", $otv)
  41.       IF($return = -1)
  42.          $hint = "Состояние чекбоксов не изменено"
  43.          HINTPOPUP($hint, "ВНИМАНИЕ")
  44.       END_IF
  45.    ELSE
  46.       $text = "Вы ничего не выбрали"
  47.    END_IF
  48. //==============================================================================
  49.    TFWRITEARR ($output_file, $Selected) // Запись списка выбранных
  50.    UNDEFINE($items)
  51.    UNDEFINE($itemsflags)
  52.    UNDEFINE($Selected)
  53.    INC($check)  // следующие 10 строк
  54. END_CYC
  55. //##############################################################################
